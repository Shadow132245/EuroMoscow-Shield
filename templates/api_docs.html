import requests

# This URL automatically adapts to your hosting
url = "{{ request.url_root }}process"

payload = {
    "code": "print('EuroMoscow System')",
    "action": "encrypt",
    "options": ["rename", "zlib", "base64"]
}

response = requests.post(url, json=payload)
